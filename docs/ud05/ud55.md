    ??? example "Ejemplo 1: Evento click"
                    
        === "index.html"

            ```html
            <!DOCTYPE html>
            <html lang="es">
                <head>
                    <meta charset="UTF-8">
                    <script src="app.js" defer></script>
                    <link rel="stylesheet" href="estilos.css">
                    <title>Click</title>
                </head>
                <body>
                    <h2>❤️ Producto favorito</h2>

                    <div class="card">
                        <p>Producto A</p>
                        <div id="fav">Añadir a favoritos</div>
                    </div>

                    <p id="msg"></p>
                </body>
            </html>
            ```
        
        === "estilos.css"

            ```css
            body {
                font-family: Arial;
                text-align: center;
            }
            .card {
                width: 200px;
                margin: auto;
                padding: 20px;
                background: #fff;
                border-radius: 10px;

            }

                background: lightblue; cursor: pointer;
            }
            ```

        === "app.js"

            ```javascript
            let elementoDiv, mensaje;

            //1. Obtener el elemento con id='fav' y almacenarlo en la variable llamada 'elementoDiv'
            elementoDiv = document.querySelector("#fav");

            //2. Obtener el elemento con id='msg' y almacenarlo en la variable llamada 'mensaje'
            mensaje = document.querySelector("#msg");

            //3. Añadir a la variable 'elementoDiv' el evento click.
            // La función del evento debe realizar las siguientes tareas:
            //  a) Asignar a la variable 'mensaje' el siguiente texto: "✔ Añadido a favoritos"
            elementoDiv.addEventListener("click", function() {
                mensaje.textContent = "✔ Añadido a favoritos";
            });
            ```

        === "SOLUCIÓN"

            ```javascript
            let productos, mensaje, coordenadas, zona;
            let i = 0, j=0;

            // ZONA MOUSE
            //1. Obtener el elemento con el id="zonaMouse" y almacenarlo en la variable 'zona'.
            zona = document.querySelector("#zonaMouse");

            //2. Obtener el elemento con el id="coordenadas" y almacenarlo en la variable 'coordenadas'.
            coordenadas = document.querySelector("#coordenadas");

            //3. Obtener todos los elementos que tengan la clase: class="producto", y almacenarlos en la variable 'productos'.
            productos = document.querySelectorAll(".producto");

            //4. Obtener el elemento con el id="mensaje" y almacenarlo en la variable 'productos'.
            mensaje = document.querySelector("#mensaje");

            // CLICK → seleccionar
            //5. Recorrer todos los productos. Para cada producto se le asignará un vento de tipo 'click'.
            // La función del evento click deberá realizar las siguientes tareas:
            //  a) Recorrer todos los productos y eliminar la clase CSS 'seleccionado'.
            //  b) Añadir a this la clase CSS 'seleccionado'.
            //  c) Asignar a la variable mensaje el texto: "Seleccionado: " + this.textContent. -> mensaje.textContent = "Seleccionado: " + this.textContent;
            while(i < productos.length){
                productos[i].addEventListener("click", function() {
                    j = 0;
                    while(j < productos.length){
                        productos[j].classList.remove("seleccionado");
                        j++;
                    }

                    this.classList.add("seleccionado");

                    mensaje.textContent = "Seleccionado: " + this.textContent;
                });

                i++;
            }


            // DBLCLICK → detalle
            //4. Recorrer todos los productos. Para cada producto se le asignará un vento de tipo 'dblclick'.
            // La función del evento DBLCLICK deberá realizar las siguientes tareas:
            //  a) Asignar a la variable 'mensaje' el texto: "🔍 Viendo detalles de " + this.textContent.
            i = 0;
            while(i < productos.length){
                productos[i].addEventListener("dblclick", function() {
                    mensaje.textContent = "🔍 Viendo detalles de " + this.textContent;
                });

                i++;
            }

            // MOUSEDOWN → inicio acción
            //5. Recorrer todos los productos. Para cada producto se le asignará un vento de tipo 'mousedown'.
            // La función del evento MOUSEDOWN deberá realizar las siguientes tareas:
            //  a) Asignar a la variable 'mensaje' el texto: "🖱️ Pulsando " + this.textContent.
            i = 0;
            while(i < productos.length){
                productos[i].addEventListener("mousedown", function() {
                    mensaje.textContent = "🖱️ Pulsando " + this.textContent;
                });

                i++;
            }

            // MOUSEUP → fin acción
            //6. Recorrer todos los productos. Para cada producto se le asignará un vento de tipo 'mouseup'.
            // La función del evento MOUSEUP deberá realizar las siguientes tareas:
            //  a) Asignar a la variable 'mensaje' el texto: "✔ Soltaste " + this.textContent.
            i = 0;
            while(i < productos.length){
                productos[i].addEventListener("mouseup", function() {
                    mensaje.textContent = "✔ Soltaste " + this.textContent;
                });

                i++;
            }

            // MOUSEOVER → resaltar
            //7. Recorrer todos los productos. Para cada producto se le asignará un vento de tipo 'mouseover'.
            // La función del evento MOUSEOVER deberá realizar las siguientes tareas:
            //  a) Asignar el color "#c7d2fe" a la propiedad background de this -> this.style.background = "#c7d2fe";
            i = 0;
            while(i < productos.length){
                productos[i].addEventListener("mouseover", function() {
                    this.style.background = "#c7d2fe";
                });

                i++;
            }

            // MOUSEOUT → quitar resaltado
            //8. Recorrer todos los productos. Para cada producto se le asignará un vento de tipo 'mouseout'.
            // La función del evento MOUSEOUT deberá realizar las siguientes tareas:
            //  a) Asignar el color "white" a la propiedad background de this.
            i = 0;
            while(i < productos.length){
                productos[i].addEventListener("mouseout", function() {
                    this.style.background = "white";
                });

                i++;
            }


            // MOUSEMOVE → coordenadas
            //9. Asignar a la variable zona un evento de tipo 'mousemove'.
            // La función del evento mousemove deberá realizar las siguientes tareas:
            //  a) Asignar las coordenadas del ratón al textContent de la variable 'coordenadas': coordenadas.textContent = "X: " + e.offsetX + " Y: " + e.offsetY;
            zona.addEventListener("mousemove", function(e) {
                coordenadas.textContent = "X: " + e.offsetX + " Y: " + e.offsetY;
            });
            ```
