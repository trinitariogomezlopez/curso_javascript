??? example "Ejercicio 1: Mueve el personaje con el teclado"
                
    === "index.html"

        ```html
        <!DOCTYPE html>
        <html lang="es">
            <head>
                <meta charset="UTF-8">
                <title>Mover con teclado</title>
                <script src="app.js" defer></script>
                <link rel="stylesheet" href="estilos.css">
            </head>
            <body>
                <h2>⌨️ Mueve el cuadrado con las flechas</h2>

                <div id="zona">
                    <div id="caja"></div>
                </div>
            </body>
        </html>
        ```
    
    === "estilos.css"

        ```css
        #zona {
            width: 400px;
            height: 300px;
            border: 2px solid black;
            position: relative;
            margin: auto;
        }

        #caja {
            width: 50px;
            height: 50px;
            background: blue;
            position: absolute;
            top: 100px;
            left: 150px;
        }
        ```

    === "app.js"

        ```javascript
        let caja = document.querySelector("#caja");
        let x = 150;
        let y = 100;

        // Emplear el evento "keydown"
        document.addEventListener("", function(e) {

            // Si la tecla pulsada es igual flecha-derecha (ArrowRight), suma 10 unidades a la variable x.
            if (e.key === "") x = x + 10;

            // Si la tecla pulsada es igual flecha-izquierda (ArrowLeft), resro 10 unidades a la variable x.

            // Si la tecla pulsada es igual flecha-arriba (ArrowUp), resto 10 unidades a la variable y.

            // Si la tecla pulsada es igual flecha-derecha (ArrowDown), suma 10 unidades a la variable y.

            //Asigno la nuevas coordenadas x-y a la caja
            caja.style.left = x + "px";
            caja.style.top = y + "px";
        });
        ```

    === "SOLUCIÓN"

        ```javascript
        let caja = document.getElementById("caja");

        let x = 150;
        let y = 100;

        document.addEventListener("keydown", function(e) {

            if (e.key === "ArrowRight") x += 10;
            if (e.key === "ArrowLeft") x -= 10;
            if (e.key === "ArrowUp") y -= 10;
            if (e.key === "ArrowDown") y += 10;

            caja.style.left = x + "px";
            caja.style.top = y + "px";
        });
        ```

        ```javascript
        let i = 0;
        let celda;

        // 1. Obtener el elemento con la clase: class = 'tablero', y almacenarlo en la variable 'tablero'
        let tablero = document.querySelector("#tablero");

        //2. Variable para comprobar si se tiene pulsado el ratón
        let pulsado = false;


        //Recorremos 100 casillas. El tablero tiene un tamaño de 10x10, que son en total 100 cuadrados.
        while(i < 100){
            //3. Creamos un elemento de tipo 'div' y lo asignamos a la variable 'celda.
            celda = document.createElement("div");

            //4. Añadimos a la variable 'celda' la clase CSS llamada 'celda'.
            celda.classList.add("celda");

            //5. Añadimos un evento del tipo 'mouseover' a la variable 'celda.
            //   Dentro del evento se realizará la siguiente comprobación:
            //      a) Si pulsado es igual a true, al objeto 'this' añadir la clase css 'pintado': this.classList.add()
            celda.addEventListener("mouseover", function() {
                if (pulsado) {
                    this.classList.add("pintado");
                }
            });

            // 6. Añadir a la variable 'tablero' la variable celda. Las celdas se añadirán al final del tablero.
            tablero.append(celda);

            i++;
        }

        // 7. Añadir un evento de ratón (mousedown), de forma que cuando se pulse el ratón a la variable 'pulsado' se le asignará el valor 'true'.
        document.addEventListener("mousedown", () => pulsado = true);

        // 8. Añadir un evento de ratón (mouseup), de forma que cuando se suelte el ratón a la variable 'pulsado' se le asignará el valor 'false'.
        document.addEventListener("mouseup", () => pulsado = false);
        ```

??? example "Ejercicio 3: Escribe y cambia color con teclado"
                
    === "index.html"

        ```html
        <!DOCTYPE html>
        <html lang="es">
            <head>
                <meta charset="UTF-8">
                <title>Texto interactivo</title>
                <script src="app.js" defer></script>
                <link rel="stylesheet" href="estilos.css">
            </head>
            <body>
                <h2>⌨️ Escribe y cambia color</h2>

                <div id="caja"></div>

                <p>Teclas:</p>
                <p>R → rojo | G → verde | B → azul</p>
            </body>
        </html>
        ```
    
    === "estilos.css"

        ```css
        #caja {
            width: 300px;
            height: 100px;
            border: 2px solid black;
            margin: auto;
            padding: 10px;
            font-size: 20px;
        }
        ```

    === "app.js"

        ```javascript
        let caja = document.getElementById("caja");

        //1. El tipo de evento a disparar será cuando se pulse una tecla del teclado (keydown)
        document.addEventListener("", function(e) {

            // 2. Mostramos las teclas que pulsamos por teclado dentro de la caja. 
            caja.textContent = caja.textContent + e.key;

            // 3. Si la tecla pulsada es igual a "r", cambiamos el color de la variable 'caja' a rojo.

            // 4. Si la tecla pulsada es igual a "r", cambiamos el color de la variable 'caja' a verde.

            // 5. Si la tecla pulsada es igual a "r", cambiamos el color de la variable 'caja' a azul.
        });
        ```

    === "SOLUCIÓN"

        ```javascript
        let caja = document.getElementById("caja");

        //1. El tipo de evento a disparar será cuando se pulse una tecla del teclado (keydown)
        document.addEventListener("keydown", function(e) {

            // 2. Mostramos las teclas que pulsamos por teclado dentro de la caja. 
            caja.textContent = caja.textContent + e.key;

            // 3. Si la tecla pulsada es igual a "r", cambiamos el color de la variable 'caja' a rojo.
            if (e.key === "r") caja.style.color = "red";

            // 4. Si la tecla pulsada es igual a "r", cambiamos el color de la variable 'caja' a verde.
            if (e.key === "g") caja.style.color = "green";

            // 5. Si la tecla pulsada es igual a "r", cambiamos el color de la variable 'caja' a azul.
            if (e.key === "b") caja.style.color = "blue";
        });
        ```
