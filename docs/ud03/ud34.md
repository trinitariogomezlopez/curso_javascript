# 4. Arrays

Un **array** (también llamado vector o arreglo) es una variable que contiene diversos valores. 

Se crea simplemente con `[]` o con `new Array()` o inicializando los elementos.

## 4.1 Métodos

Los arrays en JavaScript tienen varias propiedades y métodos interesantes que facilitan su manipulación. Entre los métodos más empleados están:

- `length`: Devuelve el número de elementos en el array.
- `sort()`: Ordena los elementos del array.
- `push()`: Agrega uno o más elementos al final del array.
- `pop()`: Elimina y retorna el último elemento del array.

    === "app.js"

        ```javascript
        document.addEventListener("DOMContentLoaded", () => {
            let tamanyoArray = 0;
            let miVector=[];
            miVector[0]=22;
            miVector[1]=12;
            miVector[2]=33;

            // Mostramos el tamaño de array
            tamanyoArray = miVector.length; //3
            document.getElementById("tamanyo").textContent = tamanyoArray;

            // Mostramos el contenido del array
            mostrarDatos();

            // Ordenamos el array y mostramos los datos ordenados
            miVector.sort();
            mostrarDatosOrdenados();

            // Añadimos un nuevo elemento al array y mostramos los datos
            miVector.push(4); // [1, 2, 3, 4]
            mostrarDatosNuevoElemento();

            // Eliminamos el último elemento al array y mostramos los datos
            miVector.pop(); // [1, 2, 3]
            mostrarDatosBorrarElemento(); 

            function mostrarDatos(){
                let resultado = 0;
                let i = 0;

                while(i < miVector.length) {
                    resultado = resultado + miVector[i] + " ";
                    i++;
                }

                document.getElementById("resultado").textContent = resultado;
            }

            function mostrarDatosOrdenados(){
                let resultado = 0;
                let i = 0;

                while(i < miVector.length) {
                    resultado = resultado + miVector[i] + " ";
                    i++;
                }

                document.getElementById("resultadosOrdenado").textContent = resultado;
            }

            function mostrarDatosNuevoElemento(){
                let resultado = 0;
                let i = 0;

                while(i < miVector.length) {
                    resultado = resultado + miVector[i] + " ";
                    i++;
                }

                document.getElementById("resultadoNuevoElemento").textContent = resultado;
            }

            function mostrarDatosBorrarElemento(){
                let resultado = 0;
                let i = 0;

                while(i < miVector.length) {
                    resultado = resultado + miVector[i] + " ";
                    i++;
                }

                document.getElementById("resultadoBorrarElemento").textContent = resultado;
            }
        });
        ```
