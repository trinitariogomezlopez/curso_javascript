# 1. Efectos con imágenes

Una imagen en HTML se inserta con una sola etiqueta. No tiene etiqueta de cierre. Los atributos más importantes son `src` (ruta de la imagen) y `alt` (texto alternativo para personas con dificultad visual o cuando la imagen no carga).

```html
<!-- La imagen más sencilla del mundo --> 
<img src="foto.jpg" alt="Mi foto de vacaciones"> 

<!-- Con tamaño definido --> 
<img src="foto.jpg" alt="Mi foto" width="300" height="200">
```

        === "app.js"

            ```javascript
            let fotoOpacidad = document.querySelector("#img-opacidad");
            let fotoByN = document.querySelector("#img-bn");
            let fotoBrillo = document.querySelector("#img-brillo");
            let fotoDesenfoque = document.querySelector("#img-desenfoque");

            /*
            IMAGEN - OPACIDAD
            */
            fotoOpacidad.addEventListener("mouseover", function () {
                fotoOpacidad.src = "img_2.jpg";
                fotoOpacidad.style.opacity = "0.4";
            });

            fotoOpacidad.addEventListener("mouseout", function () {
                fotoOpacidad.src = "img_1.jpg";
                fotoOpacidad.style.opacity = "1";
            });


            /*
            IMAGEN - BLANCO Y NEGRO
            */
            fotoByN.addEventListener("mouseover", function () {
                fotoByN.style.filter = "grayscale(100%)";
            });

            fotoByN.addEventListener("mouseout", function () {
                fotoByN.style.filter = "none";
            });


            /*
            IMAGEN - BRILLO
            */

                fotoBrillo.style.filter = "brightness(1.6)";

                fotoBrillo.style.filter = "none";
            ```

            ```javascript
            let fotoSolido = document.querySelector("#img-solido");
            let fotoSolidoDoble = document.querySelector("#img-dolido-doble");
            let fotoCircular = document.querySelector("#img-circular");
            let fotoSombraSuave = document.querySelector("#img-sombra-suave");

            IMAGEN - Borde

                fotoSolido.style.border= "6px solid #ff0000";

            fotoSolido.addEventListener("mouseout", function () {
                fotoSolido.style.border = "none";
            });

            /*
            IMAGEN - Borde Doble
            */
            fotoSolidoDoble.addEventListener("mouseover", function () {
                fotoSolidoDoble.style.border= "6px double #ff0000";
            });

            fotoSolidoDoble.addEventListener("mouseout", function () {
                fotoSolidoDoble.style.border= "none";
            });


            /*
            IMAGEN - Borde Circular
            */
            fotoCircular.addEventListener("mouseover", function () {
                fotoCircular.style.objectFit = "cover";
                fotoCircular.style.borderRadius = "50%"; // 🔥 esto hace el círculo
            });

            fotoCircular.addEventListener("mouseout", function () {
                fotoCircular.style.objectFit  = "fill";
                fotoCircular.style.borderRadius = "0";
            });

            /*
            IMAGEN - Sombra Suave
            */
            fotoSombraSuave.addEventListener("mouseover", function () {
                fotoSombraSuave.style.boxShadow  = "0px 10px 30px rgba(0,0,0,0.3)";
            });

            fotoSombraSuave.addEventListener("mouseout", function () {
                fotoSombraSuave.style.boxShadow  = "none";
            });
            ```

            ```javascript
            let imagen1 = document.querySelector("#imagen_1");
            let imagen2 = document.querySelector("#imagen_2");
            let imagen3 = document.querySelector("#imagen_3");
            let imagen4 = document.querySelector("#imagen_4");

            //----------------------------------------------
            //----------------IMAGEN 1----------------------
            imagen1.addEventListener("mouseover", function(){
                this.style.transition = "transform 0.4s ease";
                this.style.transform = "rotate(8deg)";

                this.style.transition = "transform 0.4s ease";
                this.style.transform = "rotate(0deg)";
            });


            //----------------------------------------------
            //----------------IMAGEN 2----------------------

                this.style.transition = "transform 0.4s ease";
                this.style.transform = "scale(1.15)";
            });

            imagen2.addEventListener("mouseout", function(){
                this.style.transition = "transform 0.4s ease";
                this.style.transform = "scale(1)";
            });


            //----------------------------------------------
            //----------------IMAGEN 3----------------------
            imagen3.addEventListener("mouseover", function(){
                this.style.transition = "transform 0.4s ease";
                this.style.transform = "translateY(-12px)";
            });

            imagen3.addEventListener("mouseout", function(){
                this.style.transition = "transform 0.4s ease";
                this.style.transform = "translateY(0px)";
            });


            //----------------------------------------------
            //----------------IMAGEN 4----------------------
            imagen4.addEventListener("mouseover", function(){
                this.style.transition = "transform 0.5s cubic-bezier(.68,-0.55,.27,1.55)";
                this.style.transform = "scale(1.2) rotate(-5deg)";

            imagen4.addEventListener("mouseout", function(){
                this.style.transition = "transform 0.5s cubic-bezier(.68,-0.55,.27,1.55)";
            ```
