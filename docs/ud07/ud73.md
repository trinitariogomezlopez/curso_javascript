            ```javascript
            let zMax = 10;

            let btnCorreo = document.querySelector("#btn-correo");
            let btnArchivo = document.querySelector("#btn-archivos");
            let btnCalc = document.querySelector("#btn-calc");

            btnCorreo.addEventListener("click", function(){
                crearVentanaFlotante('📧 Correo','#1a3a5e','Tienes 3 mensajes nuevos.');
            });

            btnArchivo.addEventListener("click", function(){
                crearVentanaFlotante('📁 Archivos','#2a5e3a','Carpeta Documentos vacía.');
            });

            btnCalc.addEventListener("click", function(){
                crearVentanaFlotante('🔢 Calculadora','#c45c1a','Aquí irían los botones de la calculadora.');
            });



            function crearVentanaFlotante(titulo, color, contenido) {
                let esc = document.getElementById('escritorio');
                let win = document.createElement('div');

                win.className = 'win-flotante';
                
                win.style.cssText = `top:${20 + Math.random() * 80}px; left:${20 + Math.random() * 120}px; z-index:${++zMax};`;

                win.innerHTML = `<div class="win-barra" style="background:${color};">${titulo}
                                    <button class="win-btn-x" onclick="this.closest('.win-flotante').remove()">✕</button>
                                </div>
                                <div class="win-cuerpo">${contenido}</div>`;

                esc.appendChild(win);
                
                hacerArrastrable(win, esc);

                win.addEventListener('mousedown', () => { 
                    win.style.zIndex = ++zMax; 
                });
            }

            function hacerArrastrable(el, contenedor) {
                let barra = el.querySelector('.win-barra');
                let ox, oy;

                barra.addEventListener('mousedown', e => {
                    ox = e.clientX - el.offsetLeft;
                    oy = e.clientY - el.offsetTop;

                    document.onmousemove = e2 => {
                        el.style.left = Math.max(0, e2.clientX - ox) + 'px';
                        el.style.top = Math.max(0, e2.clientY - oy) + 'px';
                    };

                    document.onmouseup = () => { 
                        document.onmousemove = document.onmouseup = null; 
                    };
                });
            }
            ```

                    icono = '⚠️';
                }

                mensaje.className = 'toast ' + tipo;
                mensaje.innerHTML = '<span class="toast-icono">'+ icono + '</span>' + texto;

                toast.appendChild(mensaje);

                setTimeout(() => {
                    mensaje.classList.add('saliendo');
                }, 3000);
            }
