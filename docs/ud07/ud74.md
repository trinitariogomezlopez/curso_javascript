# 1. Iframe

## 1.1 Ejemplos

!!! note "Ejemplos: Iframe"

    ??? example "Ejemplo 1: Insertar Mapa - Vídeo"
                    
        === "index.html"

            ```html
            <!DOCTYPE html>
            <html lang="es">

            <head>
                <meta charset="UTF-8" />
                <meta name="viewport" content="width=device-width, initial-scale=1.0" />
                <title>Filtros en Imágenes</title>
                <link rel="stylesheet" href="estilo.css">
                <script src="app.js" defer></script>
            </head>

            <body>
                <div class="ejercicio" id="ej2">
                    <h2>Ejemplo 2: Iframe con contenido dinámico</h2>
                    <p style="color:#555; margin-bottom:0.5rem;">Pulsa los botones para ver qué tipo de contenido puede mostrar un
                        iframe:</p>
                    <div class="botones-contenido">
                        <button class="btn btn-verde" id="mapa">🗺️ Mapa</button>
                        <button class="btn btn-verde" id="video">▶️ Vídeo</button>
                    </div>
                    <br />
                    <iframe id="frame"
                        width="100%" height="500px" style="border: 0;" 
                        allow="accelerometer; autoplay; clipboard-write; encrypted-media; gyroscope; picture-in-picture"
                        allowfullscreen="" 
                        loading="lazy">
                    </iframe>
                </div>
            </body>

            </html>
            ```
        
        === "app.js"

            ```javascript
            let frame = document.querySelector("#frame");
            let mapa = document.querySelector("#mapa");
            let video = document.querySelector("#video");

            mapa.addEventListener("click", function(){
                frame.src = "https://www.google.com/maps/embed?pb=!1m18!1m12!1m3!1d11933.923265420306!2d-0.8918448407816243!3d38.12192910039537!2m3!1f0!2f0!3f0!3m2!1i1024!2i768!4f13.1!3m3!1m2!1s0xd63a2e98b5af579%3A0x85c0551478e2c23d!2s03360%20Callosa%20de%20Segura%2C%20Alicante!5e1!3m2!1ses!2ses!4v1777823129955!5m2!1ses!2ses";
            });

            video.addEventListener("click", function(){
                frame.src = 'https://www.youtube.com/embed/oznULIjXqEI?si=DwJYqX_3MsY5jF77';
            });
            ```
