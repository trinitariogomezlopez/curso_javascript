            ```javascript
            /*
                ENUNCIADO: Cuenta los alumnos, años y cursos desde el 0 hasta el valor indicado 
            */
            let iniciar = document.querySelector("#btn-contar");

            iniciar.addEventListener("click", function () {
                let alumnos = document.getElementById('c1');
                let cursos = document.getElementById('c2');
                let anyios = document.getElementById('c3')
                
                alumnos.textContent = '0';
                anyios.textContent = '0';
                cursos.textContent = '0';

                contarHasta(alumnos, 1247, 2000);
                contarHasta(cursos, 38, 2000);
                contarHasta(anyios, 15, 2000);
            });

            function contarHasta(elemento, final, duracion) {
                let inicio = Date.now();

                let tick = () => {
                    let progreso = Math.min((Date.now() - inicio) / duracion, 1);
                    let ease = progreso * (2 - progreso); /* HACE QUE SE  DESACELERE POR EL FINAL*/

                    elemento.textContent = Math.round(ease * final); 

                    if (progreso < 1) {
                        requestAnimationFrame(tick);
                    }
                };
                requestAnimationFrame(tick);
            }
            ```

        === "app.js"

            ```javascript
            /*
                ENUNCIADO 1: Añadir un evento click al botón 'frase 1' para que muestre una frase aplicando el efecto máquina de escribir.
            */

            /*
                ENUNCIADO 2: Añadir un evento click al botón 'frase 2' para que muestre una frase aplicando el efecto máquina de escribir.
            */

            /*
                ENUNCIADO 3: Añadir un evento click al botón 'borrar' para que borre la frase.
            */
            let idTimer = null;

            let frase1 = document.querySelector("#frase1");
            let frase2 = document.querySelector("#frase2");
            let borrar = document.querySelector("#borrar");

            //ENUNCIADO 1: Añadir un evento click al botón 'frase 1' para que muestre una frase aplicando el efecto máquina de escribir.
            frase1.addEventListener("click", function(){
                escribir("¡Bienvenido a nuestra página web!");
            });

            //ENUNCIADO 2: Añadir un evento click al botón 'frase 2' para que muestre una frase aplicando el efecto máquina de escribir.
            frase2.addEventListener("click", function(){
                escribir("Aprendemos HTML y CSS paso a paso.");
            });
                
            function escribir(texto) {
                let i = 0;
                let escribir = document.getElementById('escribir');

                clearInterval(idTimer);

                escribir.textContent = '';
                escribir.classList.add('escribiendo');
                
                idTimer = setInterval(() => {
                    escribir.textContent += texto[i++];
                    if (i >= texto.length) { 
                        clearInterval(idTimer); 
                        escribir.classList.remove('escribiendo'); 
                    }
                }, 60);
            }

            // ENUNCIADO 3: Añadir un evento click al botón 'borrar' para que borre la frase.
            borrar.addEventListener("click", function () {
                let escribir = document.getElementById('escribir');

                clearInterval(idTimer);

                escribir.classList.add('escribiendo');

                idTimer = setInterval(() => {
                    escribir.textContent = escribir.textContent.slice(0, -1);
                    if (!escribir.textContent) {
                        clearInterval(idTimer); 
                        escribir.classList.remove('escribiendo'); 
                    }
                }, 30);
            });
            ```
